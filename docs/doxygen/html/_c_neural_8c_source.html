219float CNeural_activation(float input, string af) {
220 if (strcmp(af, "none") == 0) return input;
221
222 if (strcmp(af, "sigmoid") == 0) {
223 return 1 / (1 + expf(-input));
224 }
225 if (strcmp(af, "tanh") == 0) {
226 return tanhf(input);
227 }
228 if (strcmp(af, "relu") == 0) {
229 return fmaxf(0, input);
230 }
231 printf("Warning: Unknown activation function. Training results might not be optimal!\n");
232 printf("Defaulting to ReLu\n");
233 return fmaxf(0, input);
234}

245float CNeural_loss(float predicted[], float actual[], int outputShape, string lfn) {
246 if (strcmp(lfn, "mse") == 0) { // Mean Squared Error
247 float sum = 0;
248 for (int i = 0; i < outputShape; i++) {
249 sum += powf(predicted[i] - actual[i], 2);
250 }
251 return sum;
252 }
253 if (strcmp(lfn, "mae") == 0) { // Mean Average Error
254 float sum = 0;
255 for (int i = 0; i < outputShape; i++) {
256 sum += fabsf(predicted[i] - actual[i]);
257 }
258 return sum;
259 }
260
261 printf("Warning: Unknown loss function. Training results might not be optimal!\n");
262 printf("Defaulting to MSE\n");
263 float sum = 0;
264 for (int i = 0; i < outputShape; i++) {
265 sum += powf(predicted[i] - actual[i], 2);
266 }
267 return sum;
268}
